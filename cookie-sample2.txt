
<script src="https://cdn.jsdelivr.net/npm/js-cookie@rc/dist/js.cookie.min.js"></script>

 
<script>
    var popupCookie = Cookies.get('openedPopup');

    $('#popup').hide();

    if(!popupCookie) {
        $('#popup').show();
    }


    $('.close-popup').on('click', function() {
        Cookies.set('openedPopup', true);

        $('#popup').hide();
    });
</script>
